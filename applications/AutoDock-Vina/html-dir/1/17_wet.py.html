96   	def rotatePoint(pt,m,ax):
97   	    # From Ludo
98   	    x=pt[0]
99   	    y=pt[1]
100  	    z=pt[2]
101  	    u=ax[0]
102  	    v=ax[1]
103  	    w=ax[2]
104  	    ux=u*x
105  	    uy=u*y
106  	    uz=u*z
107  	    vx=v*x
108  	    vy=v*y
109  	    vz=v*z
110  	    wx=w*x
111  	    wy=w*y
112  	    wz=w*z
113  	    sa=sin(ax[3])
114  	    ca=cos(ax[3])
115  	    pt[0]=(u*(ux+vy+wz)+(x*(v*v+w*w)-u*(vy+wz))*ca+(-wy+vz)*sa)+ m[0]
116  	    pt[1]=(v*(ux+vy+wz)+(y*(u*u+w*w)-v*(ux+wz))*ca+(wx-uz)*sa)+ m[1]
117  	    pt[2]=(w*(ux+vy+wz)+(z*(u*u+v*v)-w*(ux+vy))*ca+(-vx+uy)*sa)+ m[2]
118  	    return pt

148  	def hydro (atom1, atom2, atom3 = None, spacing = 3):
149  	    # INFO   : place a water atom W at given distance from an atom
150  	    # INPUT  : (a) two pdb lines for N-H; (b) three pdb lines for -[N|O]- acceptor, -O-H donors; the spacing distance between the water and the atom
151  	    # OUTPUT : one pdb line 
152  	    #
153  	    # Synopsys:
154  	    #
155  	    # -C-NH-C  => atom1 = N, atom2 = H
156  	    # -C-N=C   => atom1 = C, atom2 = N, atom3 = C
157  	    # -C-O-H   => atom1 = C, atom2 = O, atom3 = H
158  	
159  	    if atom2.split()[-1] == "HD":
160  	        spacing -= 1 # to put the W at 3A from the N-H
161  	    index = 99 
162  	    coord2 = atom_coord(atom2)
163  	    x=(coord2[0])
164  	    y=(coord2[1])
165  	    z=(coord2[2])
166  	    chain="X"
167  	
168  	
169  	    if atom3:
170  	        atom4 = mean_pdb(atom1, atom3)
171  	        vec_module = dist(atom2, atom4)
172  	        coord1 = atom_coord(atom4) 
173  	    else:
174  	        coord1 = atom_coord(atom1)
175  	        vec_module = dist(atom1, atom2)
176  	
177  	    alpha = math.acos((coord2[0]-coord1[0])/vec_module)    # x-axis angle
178  	    beta  = math.acos((coord2[1]-coord1[1])/vec_module)    # y-axis angle
179  	    gamma = math.acos((coord2[2]-coord1[2])/vec_module)    # z-axis angle
180  	    wat_x = spacing*math.cos(alpha)+x
181  	    wat_y = spacing*math.cos(beta)+y
182  	    wat_z = spacing*math.cos(gamma)+z
183  	
184  	    wet = "%s%5d %2s   %3s %1s%4d    %8.3f%8.3f%8.3f  1.00 10.00     %1.3f %1s\n" % (keyw, index, ATYPE, residue, chain, index, wat_x, wat_y, wat_z, pcharge, ATYPE)
185  	    return wet

244  	    """
245  	    identify all the atoms in "structure" that are @bond_dist from "atom"
246  	
247  	    NOTE: this should be made with a lookup table!
248  	    """
249  	    bound_list = []
250  	    tolerance = 0
251  	
252  	    DEBUG = 0

254  	        print("Finding mates for ", atom)

258  	    elif atype == "S" or atype == "SA":
259  	        bond_dist = 1.95
260  	    for candidate in structure:
261  	        if candidate == atom or candidate == exclude:
262  	            pass
263  	        else:
264  	            if candidate[0:4] == "ATOM" or candidate[0:6] == "HETATM":
265  	                c_atype = candidate.split()[-1]
266  	                if c_atype == "SA" or c_atype == "S" or c_atype  == "P":
267  	                    if not atype == "HD":
268  	                        tolerance = .20
269  	                    else:

272  	                    tolerance = -.5
273  	                else:
274  	                    tolerance = 0
275  	
276  	                #print candidate.strip(), tolerance+bond_dist
277  	                if dist(atom, candidate) <= bond_dist + tolerance:
278  	                    if not candidate in bound_list:
279  	                        bound_list.append(candidate)
280  	                        #print candidate,
281  	                        #print tolerance+bond_dist
282  	                else:
283  	                    pass
284  	    if len(bound_list) > 0:
285  	        # Extra clean-up required for planar structures
286  	        # where the HD lies between too many interested atoms...
287  	        if atype == "HD":
288  	            min = 999

374  	def coplanar(plane, structure, reference, tolerance = .2):
375  	    # identify all the atoms in "structure" that are co-planar with "plane"
376  	    # the dot product between the point and the plane must be ~= 0
377  	    
378  	    coplane_list = []
379  	    for atom in structure:
380  	        position = vector(reference, atom)
381  	        if dot(plane, position) <= tolerance:
382  	            coplane_list.append(atom)
383  	    return coplane_list

453  	def gpfminmax(gpf):
454  	    # INPUT : gpf_file
455  	    # OUTPUT: box coordinates (x,y,z), (X,Y,Z)
456  	    """ Return max/min coordinates of
457  	    the box described in the GPF"""
458  	    if not GPF:
459  	        return True
460  	    #print gpf    
461  	    file = open(gpf, 'r')
462  	    lines = file.readlines()
463  	    file.close()
464  	
465  	    for line in lines:
466  	        tmp=line.split()
467  	        if tmp[0] == "gridcenter":
468  	            center_x = float(tmp[1])
469  	            center_y = float(tmp[2])
470  	            center_z = float(tmp[3])
471  	        if tmp[0] == "npts":
472  	            pts_x = float(tmp[1])
473  	            pts_y = float(tmp[2])
474  	            pts_z = float(tmp[3])
475  	        if tmp[0] == "spacing":
476  	            res = float(tmp[1])
477  	
478  	    step_x = pts_x/2 * res
479  	    step_y = pts_y/2 * res
480  	    step_z = pts_z/2 * res
481  	
482  	    x_min = center_x - step_x
483  	    x_max = center_x + step_x
484  	
485  	    y_min = center_y - step_y
486  	    y_max = center_y + step_y
487  	
488  	    z_min = center_z - step_z
489  	    z_max = center_z + step_z
490  	
491  	    print(" - using the GPF box filter [ %s ]"% gpf)
492  	
493  	    return [x_min, y_min, z_min], [x_max, y_max, z_max]
