18   	class Vina:
19   	    def __init__(self, sf_name='vina', cpu=0, seed=0, no_refine=False, verbosity=1):
20   	        """Initialize a Vina object.
21   	
22   	        Args:
23   	            sf_name (str): Scoring function name to use (Vina or ad4) (default: vina)
24   	            cpu (int): Number of CPU to use (default: 0; use all of them)
25   	            seed (int): Random seed (default: 0; ramdomly choosed)
26   	            no_refine (boolean): when receptor is provided, do not use explicit receptor atoms
27   	                (instead of precalculated grids) for: (1) local optimization and scoring after docking,
28   	                (2) --local_only jobs, and (3) --score_only jobs (default: False)
29   	            verbosity (int): verbosity 0: not output, 1: normal, 2: verbose (default: 1; some output)
30   	
31   	        """
32   	        sf_name = sf_name.lower()
33   	        if not sf_name in ('vina', 'vinardo', 'ad4'):
34   	            raise ValueError('Error: Scoring function %s not recognized. (only vina, vinardo or ad4)' % sf_name)
35   	
36   	        self._vina = _Vina(sf_name, cpu, seed, verbosity, no_refine)
37   	        
38   	        self._sf_name = sf_name
39   	        if sf_name == 'vina':
40   	            self._weights = (-0.035579, -0.005156, 0.840245, -0.035069, -0.587439, 50, 0.05846)
41   	        elif sf_name == 'vinardo':
42   	            self._weights = (-0.045, 0.8, -0.035, -0.6, 50, 0.05846)
43   	        else:
44   	            self._weights = (0.1662, 0.1209, 0.1406, 0.1322, 50)
45   	        self._rigid_receptor = None
46   	        self._flex_receptor = None
47   	        self._ligands = None
48   	        self._center = None
49   	        self._box_size = None
50   	        self._spacing = None
51   	        self._even_nelements = True
52   	        if sf_name in ('vina', 'vinardo'):
53   	            self._no_refine = no_refine
54   	        else:
55   	            self._no_refine = False
56   	        self._seed = self._vina.seed()

63   	        """

126  	    def set_receptor(self, rigid_pdbqt_filename=None, flex_pdbqt_filename=None):
127  	        """Set receptor.
128  	
129  	        Args:
130  	            rigid_pdbqt_filename (str): rigid pdbqt receptor filename (default: None)
131  	            flex_pdbqt_filename (str): flexible residues pdbqt filename (default: None)
132  	
133  	        """
134  	        if rigid_pdbqt_filename is None and flex_pdbqt_filename is None:
135  	            raise ValueError('Error: No (rigid) receptor or flexible residues were specified')
136  	
137  	        # For the rigid part of the receptor
138  	        if rigid_pdbqt_filename is not None:
139  	            if not os.path.exists(rigid_pdbqt_filename):
140  	                raise RuntimeError('Error: file %s does not exist.' % rigid_pdbqt_filename)
141  	            _, extension = os.path.splitext(rigid_pdbqt_filename)
142  	            if not extension == '.pdbqt':
143  	                raise TypeError('Error: Vina requires a PDBQT file for the (rigid) receptor.')
144  	
145  	        # For the flex part of the receptor
146  	        if flex_pdbqt_filename is not None:
147  	            if not os.path.exists(flex_pdbqt_filename):
148  	                raise RuntimeError('Error: file %s does not exist.' % flex_pdbqt_filename)
149  	            _, extension = os.path.splitext(flex_pdbqt_filename)
150  	            if not extension == '.pdbqt':
151  	                raise TypeError('Error: Vina requires a PDBQT file for the (flex) receptor.')
152  	
153  	        if rigid_pdbqt_filename is not None:
154  	            if flex_pdbqt_filename is not None:
155  	                self._vina.set_receptor(rigid_pdbqt_filename, flex_pdbqt_filename)
156  	            else:
157  	                self._vina.set_receptor(rigid_pdbqt_filename)
158  	        else:
159  	            self._vina.set_receptor('', flex_pdbqt_filename)
160  	
161  	        self._rigid_receptor = rigid_pdbqt_filename
162  	        self._flex_receptor = flex_pdbqt_filename

232  	    def compute_vina_maps(self, center, box_size, spacing=0.375, force_even_voxels=False):
233  	        """Compute affinity maps using Vina scoring function.
234  	
235  	        Args:
236  	            center (list): center position
237  	            box_size (list): size of the box in Angstrom
238  	            spacing (float): grid spacing (default: 0.375)
239  	            force_even_voxels (boolean): Force the number of voxels (NPTS/NELEMENTS) to be an even number
240  	                (and forcing the number of grid points to be odd) (default: False)
241  	
242  	        """
243  	        if len(center) != 3:
244  	            raise ValueError('Error: center of the box needs to be defined by (x, y, z) in Angstrom.')
245  	        elif len(box_size) != 3:
246  	            raise ValueError('Error: box size needs to be defined by (a, b, c) in Angstrom.')
247  	        elif not all([i > 0 for i in box_size]):
248  	            raise ValueError('Error: box dimensions are required to be positive.')
249  	        elif spacing <= 0:
250  	            raise ValueError('Error: spacing should be greater than zero.')
251  	
252  	        x, y, z = center
253  	        a, b, c = box_size
254  	
255  	        self._vina.compute_vina_maps(x, y, z, a, b, c, spacing, force_even_voxels)
256  	
257  	        self._center = center
258  	        self._box_size = box_size
259  	        self._spacing = spacing
260  	        self._voxels = np.ceil(np.array(box_size) / self._spacing).astype(np.int)
261  	
262  	        # Necessary step to know if we can write maps or not later
263  	        if force_even_voxels:
264  	            self._voxels[0] += int(self._voxels[0] % 2 == 1)
265  	            self._voxels[1] += int(self._voxels[1] % 2 == 1)
266  	            self._voxels[2] += int(self._voxels[2] % 2 == 1)

282  	        """Write affinity maps.
283  	
284  	        Args:
285  	            map_prefix_filename (str): affinity map pathname (path directory + prefix)
286  	            gpf_filename (str): grid protein filename (default: NULL)
287  	            fld_filename (str): fld filename (default: NULL)

294  	        elif not overwrite:
295  	            existing_maps = glob.glob('%s.*.map' % map_prefix_filename)
296  	            if existing_maps:
297  	                raise RuntimeError('Error: Cannot overwrite existing affinity maps (%s)' % existing_maps)
298  	
299  	        if any(self._voxels % 2 == 1):
300  	            error_msg = 'Error: Can\'t write maps. Number of voxels (NPTS/NELEMENTS) is odd: %s.' % self._voxels
301  	            error_msg += ' Set \'force_even_voxels\' to True when computing vina maps.'
302  	            raise RuntimeError(error_msg)
303  	
304  	        self._vina.write_maps(map_prefix_filename, gpf_filename, fld_filename, receptor_filename)

348  	    def poses(self, n_poses=9, energy_range=3.0, coordinates_only=False):
349  	        """Get poses from docking.
350  	
351  	        Args:
352  	            n_pose (int): number of poses to retrieve (default: 9)
353  	            energy_range (float): maximum energy difference from best pose (default: 3.0 kcal/mol)
354  	            coordinates_only (bool): return coordinates for each pose only
355  	
356  	        Returns:
357  	            str/ndarray: PDBQT file string or Array of coordinates of poses if coordinates_only=True
358  	
359  	        """
360  	        if n_poses <= 0:
361  	            raise ValueError('Error: number of poses written must be greater than zero.')
362  	        elif energy_range <= 0:
363  	            raise ValueError('Error: energy range must be greater than zero.')
364  	        
365  	        if coordinates_only:
366  	            # Dirty hack to get the coordinates from C++, it is not advised to have vector<vector<vector<double>>>
367  	            coordinates = np.array(self._vina.get_poses_coordinates(n_poses, energy_range))
368  	            coordinates = coordinates.reshape((coordinates.shape[0], coordinates.shape[1] // 3, 3))
369  	            coordinates = np.around(coordinates, decimals=3)
370  	            return coordinates
371  	        else:
372  	            return self._vina.get_poses(n_poses, energy_range)

420  	    def optimize(self, max_steps=0):
421  	        """Quick local BFGS energy optimization.
422  	
423  	        Args:
424  	            max_steps (int): Maximum number of local minimization steps (default: 0). When max_steps is equal to 0,
425  	                             the maximum number of steps will be equal to (25 + num_movable_atoms) / 3).
426  	
427  	        Returns:
428  	            nadarray: Array of energies from optimized pose.
429  	
430  	            Vina/Vinardo FF:
431  	                columns=[total, lig_inter, flex_inter, other_inter, flex_intra, lig_intra, torsions, lig_intra best pose]
432  	            
433  	            AutoDock FF:
434  	                columns=[total, lig_inter, flex_inter, other_inter, flex_intra, lig_intra, torsions, -lig_intra]
435  	
436  	        """
437  	        if max_steps < 0:
438  	            raise ValueError('Error: max_steps cannot be negative.')
439  	
440  	        # It does not make sense to report energies with a precision higher than 3
441  	        # since the coordinates precision is 3.
442  	        energies = np.around(self._vina.optimize(max_steps), decimals=3)
443  	        return energies

445  	    def dock(self, exhaustiveness=8, n_poses=20, min_rmsd=1.0, max_evals=0):
446  	        """Docking: global search optimization.
447  	
448  	        Args:
449  	            exhaustiveness (int): Number of MC run (default: 8)
450  	            n_poses (int): number of pose to generate (default: 20)
451  	            min_rmsd (float): minimum RMSD difference between poses (default: 1.0 Ansgtrom)
452  	            max_evals (int): Maximum number of evaluation (default: 0; use heuristics rules)
453  	
454  	        """
455  	        if exhaustiveness <= 0:
456  	            raise ValueError('Error: exhaustiveness must be 1 or greater.')
457  	        elif n_poses <= 0:
458  	            raise ValueError('Error: number of poses to generate must be greater than zero.')
459  	        elif min_rmsd <= 0:
460  	            raise ValueError('Error: minimal RMSD must be greater than zero.')
461  	        elif max_evals < 0:
462  	            raise ValueError('Error: maximum evaluations must be positive.')
463  	
464  	        self._vina.global_search(exhaustiveness, n_poses, min_rmsd, max_evals)
465
