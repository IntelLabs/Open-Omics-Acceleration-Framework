248  	def minmax(models):
249  	    # INPUT : pdb-like line
250  	    # OUTPUT: coordinates (x,y,z), (X,Y,Z)
251  	    """ Return max/min coordinates of
252  	    the box including the given atoms list"""
253  	
254  	    x_min, y_min, z_min =  999999999,  9999999999,  9999999999
255  	    x_max, y_max, z_max = -999999999, -9999999999, -9999999999
256  	    for pose in models:
257  	        for line in pose:
258  	            if line[0:4] == "ATOM" or line[0:6] == "HETATM":
259  	                atom = coord(line)
260  	                # min
261  	                if atom[0] < x_min:
262  	                    x_min = atom[0]
263  	                if atom[1] < y_min:
264  	                    y_min = atom[1]
265  	                if atom[2] < z_min:
266  	                    z_min = atom[2]
267  	                # max
268  	                if atom[0] > x_max:
269  	                    x_max = atom[0]
270  	                if atom[1] > y_max:
271  	                    y_max = atom[1]
272  	                if atom[2] > z_max:
273  	                    z_max = atom[2]
274  	    return [x_min, y_min, z_min], [x_max, y_max, z_max]

447  	    hcutoff = cutoff - 0.7
448  	    count = 0
449  	    for pose in model:
450  	        waters = []
451  	        mates = []
452  	        idx = model.index(pose)
453  	        for entry in pose:

455  	                if atom_type(entry) == "W":

471  	                    max = cutoff

490  	def average(list):
491  	    # INPUT : pdb atom/het lines
492  	    # OUTPUT: average coordinates
493  	
494  	    total = [ 0., 0., 0.]
495  	    for atom in list:
496  	        a_coords = coord(atom)
497  	        total[0] += a_coords[0]
498  	        total[1] += a_coords[1]
499  	        total[2] += a_coords[2]
500  	
501  	    avg = [ total[0]/len(list), total[1]/len(list), total[2]/len(list) ]
502  	    return avg

515  	def cluster_waters(model, cutoff = OVERLAP_DISTANCE):
516  	    # TODO decide if report the pop size as:
517  	    #    W3+
518  	    #   b-factor
519  	    #   partial charge   <-
520  	
521  	    count = 0
522  	    for pose in model:
523  	        waters = []
524  	        candidates = []
525  	        idx = model.index(pose)
526  	        for entry in pose:
527  	            if entry[0:4] == "ATOM" or entry[0:6] == "HETATM":
528  	                if pdbqt_atom_type(entry) == "W":
529  	                    waters.append(entry)
530  	                    pose.remove(entry)
531  	        print(" [cluster: found %d waters]" % len(waters))
532  	    
533  	        for w in waters:
534  	            pop = [w]
535  	            for mate in waters:
536  	                if not mate == w:
537  	                    if dist(mate, w) <= cutoff:
538  	                        pop.append(mate)
539  	                        waters.remove(mate)
540  	                        waters.remove(w)
541  	            if len(pop)> 0:
542  	                candidates.append(pop)
543  	    
544  	        for pop in candidates:
545  	            count = len(pop)
546  	            avg = average(pop)
547  	            avg_atom = pdbatom(avg, atype = "W", pcharge = count)
548  	            pose.append(avg_atom)
549  	        model[idx] = pose
550  	
551  	    if count > 1:
552  	        if not QUIET: print("\t\t\t [ %d clustered ]" % count)
553  	    else:
554  	        if not QUIET: print("\t\t\t [ none ]") 
555  	    return model

611  	    real = len(exp_list)
612  	    predicted = len(calc_list)
613  	    if not QUIET: print("real/predicted", real,predicted)
614  	    
615  	    found = 0
616  	
617  	    min = 9999999
618  	    stack = []
619  	    invented = 0
620  	    if real > predicted:
621  	        missed = real-predicted
622  	    if real < predicted:
623  	        invented = predicted-real
624  	    for e in exp_list:
625  	        for w in calc_list:
626  	            if (not e in stack) and (not w in stack):
627  	                measure = dist(w,e)
628  	                if measure <= cutoff:
629  	                    found += 1
630  	                    stack.append(w)
631  	                    stack.append(e)
632  	                if measure < min:
633  	                    min = measure
634  	    missed = real-found
635  	    success = (float(found)/float(real))*100
636  	    return (found, success, missed, invented)

641  	    real = len(exp_list)
642  	    predicted = len(calc_list)
643  	    if DEBUG: print("confirm_predictions>real/predicted", real,predicted)
644  	    
645  	    found = 0
646  	    distance_log = []
647  	    for w in calc_list:
648  	        for e in exp_list:
649  	            d = dist(w,e)
650  	            if d < cutoff:
651  	                found += 1
652  	                distance_log.append([w, e, d])
653  	                break
654  	    return [found, distance_log]

661  	    spacing = float(data[3].split()[1])
662  	
663  	    pts = data[4].split()[1:]
664  	    for i in range(len(pts)):
665  	        pts[i] = int(pts[i])+1
666  	
667  	    center = data[5].split()[1:]
668  	    for i in range(len(pts)):
669  	        center[i] = float(center[i])
670  	
671  	    data = data[6:]
672  	    for i in range(len(data)):
673  	        data[i] = float(data[i])
674  	
675  	    step = []
676  	    step.append(pts[0]/2 * spacing)
677  	    step.append(pts[1]/2 * spacing)
678  	    step.append(pts[2]/2 * spacing)
679  	
680  	    min, max = [], []
681  	
682  	    min.append(center[0]-step[0])
683  	    min.append(center[1]-step[1])
684  	    min.append(center[2]-step[2])
685  	
686  	    max.append(center[0]+step[0])
687  	    max.append(center[1]+step[1])
688  	    max.append(center[2]+step[2])
689  	
690  	    data = array(data, 'f').reshape(pts[2], pts[1], pts[0])
691  	
692  	    data = data.reshape(pts[2], pts[1], pts[0])
693  	    return { "values" : data, "spacing" : spacing, 'pts': pts, 'center' : center, 'min' : min, 'max' : max}

697  	def getpoints(mapdata, coords=None, distance=None):
698  	    data = mapdata['values']
699  	    min = mapdata['min']
700  	    max = mapdata['max']
701  	    spacing = mapdata['spacing']
702  	    pts = mapdata['pts']
703  	    harvesting = []
704  	    
705  	    if not distance:
706  	        distace = 1.0 # default 1A distance scanning
707  	
708  	    if not coords and not distance:
709  	        coords = [3., 4., 5.]
710  	        distance = 1.0
711  	
712  	    pt_scan = int( round(float(distance)/(spacing)) )
713  	    if pt_scan == 0: pt_scan = 1 # at least one point around
714  	    if DEBUG: print("Grid range: %2.2f [ +/- %d points : %2.2f ]" % (distance,pt_scan, pt_scan*spacing), end=' ') 
715  	    best = 999
716  	
717  	    if (coords[0] < max[0]) and (coords[0] > min[0]):
718  	        if (coords[1] < max[1]) and (coords[1] > min[1]):
719  	            if (coords[2] < max[2]) and (coords[2] > min[2]):
720  	                z_pt = int(round((coords[2] - min[2])/spacing))
721  	                y_pt = int(round((coords[1] - min[1])/spacing))
722  	                x_pt = int(round((coords[0] - min[0])/spacing))
723  	
724  	                for x_ofs in range(-pt_scan, pt_scan+1): 
725  	                    x = x_pt + x_ofs
726  	                    if x < 0 :
727  	                        harvesting.append(0)
728  	                        break
729  	                    for y_ofs in range(-pt_scan, pt_scan+1): 
730  	                        y = y_pt + y_ofs
731  	                        if y < 0 :
732  	                            harvesting.append(0)
733  	                            break
734  	                        for z_ofs in range(-pt_scan, pt_scan+1): 
735  	                            z = z_pt + z_ofs
736  	                            if z < 0 :
737  	                                harvesting.append(0)
738  	                                break
739  	                            harvesting.append( data[z,y,x] )
740  	                            if data[z,y,x] < best:
741  	                                best = data[z,y,x]
742  	            if DEBUG: print(" %d pts [ best: %2.2f ]" % (len(harvesting), best), end=' ')
743  	            return harvesting, best
744  	    return False
